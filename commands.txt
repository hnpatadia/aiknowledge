=========================================== MCP Server
dotnet new web -n DemoMcp
cd DemoMcp

dotnet add package ModelContextProtocol.AspNetCore

using ModelContextProtocol.Server;

var builder = WebApplication.CreateBuilder(args);

builder.Services
    .AddMcpServer()
    .WithHttpTransport()
    .WithToolsFromAssembly();

var app = builder.Build();

app.MapMcp();

app.Run();
